  ${EINA_INCLUDE_DIRS}
  ${EVAS_INCLUDE_DIRS}
  ${EFL_INCLUDE_DIRS}
  ${EIO_INCLUDE_DIRS}
  ${EET_INCLUDE_DIRS}
  ${EFREET_INCLUDE_DIRS}
  ${ECORE_INCLUDE_DIRS}

  ../widgets/
  )
